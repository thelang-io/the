#include <stdbool.h>

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct str {
  char *d;
  size_t l;
};

void *alloc (size_t);
struct str char_repeat (char, uint32_t);
struct str str_alloc (const char *);
bool str_empty (struct str);

void *alloc (size_t l) {
  void *r = malloc(l);
  if (r == NULL) {
    fprintf(stderr, "Error: failed to allocate %zu bytes" THE_EOL, l);
    exit(EXIT_FAILURE);
  }
  return r;
}
struct str char_repeat (char c, uint32_t k) {
  if (k == 0) return str_alloc("");
  size_t l = (size_t) k;
  char *d = alloc(l);
  size_t i = 0;
  while (i < l) d[i++] = c;
  return (struct str) {d, l};
}
struct str str_alloc (const char *r) {
  size_t l = strlen(r);
  char *d = alloc(l);
  memcpy(d, r, l);
  return (struct str) {d, l};
}
bool str_empty (struct str s) {
  bool r = s.l == 0;
  free(s.d);
  return r;
}

int main () {
  str_empty(str_alloc(""));
  str_empty(char_repeat('\0', 1));
  str_empty(str_alloc("t"));
  str_empty(str_alloc("test"));
  str_empty(char_repeat('\0', 0xFFFFFFFF));
}
