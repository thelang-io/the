void *alloc (size_t l) {
  void *r = malloc(l);
  if (r == NULL) {
    fprintf(stderr, "Error: failed to allocate %zu bytes" THE_EOL, l);
    exit(EXIT_FAILURE);
  }
  return r;
}
struct str str_alloc (const char *r) {
  size_t l = strlen(r);
  char *d = alloc(l);
  memcpy(d, r, l);
  return (struct str) {d, l};
}
void str_free (struct str s) {
  free(s.d);
}
bool fs_isDirectorySync (struct str s) {
  char *c = alloc(s.l + 1);
  memcpy(c, s.d, s.l);
  c[s.l] = '\0';
  struct stat r;
  bool b = false;
  if (stat(c, &r) == 0) {
    b = S_ISDIR(r.st_mode);
  }
  free(c);
  str_free(s);
  return b;
}

  fs_isDirectorySync(str_alloc("test/fixtures/test.txt"));
  fs_isDirectorySync(str_alloc("test/fixtures"));
}
