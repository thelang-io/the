void fs_linkSync (struct str s1, struct str s2) {
  char *c1 = str_cstr(s1);
  char *c2 = str_cstr(s2);
  #ifdef THE_OS_WINDOWS
    for (size_t i = 0; i < s1.l; i++) c1[i] = (c1[i] == '/' ? '\\' : c1[i]);

    DWORD flags = 0;

    bool r = CreateSymbolicLink(c2, c1, flags);
