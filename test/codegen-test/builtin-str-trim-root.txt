#include <ctype.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct str {
  char *d;
  size_t l;
};

void *alloc (size_t);
struct str str_alloc (const char *);
void str_free (struct str);
struct str str_trim (struct str);

void *alloc (size_t l) {
  void *r = malloc(l);
  if (r == NULL) {
    fprintf(stderr, "Error: failed to allocate %zu bytes" THE_EOL, l);
    exit(EXIT_FAILURE);
  }
  return r;
}
struct str str_alloc (const char *r) {
  size_t l = strlen(r);
  char *d = alloc(l);
  memcpy(d, r, l);
  return (struct str) {d, l};
}
void str_free (struct str s) {
  free(s.d);
}
struct str str_trim (struct str s) {
  if (s.l == 0) return s;
  size_t i = 0;
  size_t j = s.l;
  while (i < s.l && isspace(s.d[i])) i++;
  while (j >= 0 && isspace(s.d[j - 1])) {
    j--;
    if (j == 0) break;
  }
  if (i >= j) {
    free(s.d);
    return str_alloc("");
  }
  size_t l = j - i;
  char *r = alloc(l);
  for (size_t k = 0; k < l;) r[k++] = s.d[i++];
  free(s.d);
  return (struct str) {r, l};
}

int main () {
  str_free((struct str) str_trim(str_alloc("")));
}
