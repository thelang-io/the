void *alloc (size_t l) {
  void *r = malloc(l);
  if (r == NULL) {
    fprintf(stderr, "Error: failed to allocate %zu bytes" THE_EOL, l);
    exit(EXIT_FAILURE);
  }
  return r;
}
struct str path_tempDirectory () {
  char *f;
  if (
    (f = getenv("TMPDIR")) == NULL &&
    (f = getenv("TMP")) == NULL &&
    (f = getenv("TEMP")) == NULL &&
    (f = getenv("TEMPDIR")) == NULL
  ) {
    f = "/tmp";
  }
  char *fmt = f[strlen(f) - 1] == '/' ? "%sXXXXXX" : "%s/XXXXXX";
  size_t l = snprintf(NULL, 0, fmt, f);
  char *t = alloc(l + 1);
  sprintf(t, fmt, f);
  char *d = mkdtemp(t);
  if (d == NULL) {
    fprintf(stderr, "Error: failed to create temporary directory" THE_EOL);
    exit(EXIT_FAILURE);
  }
  struct str r = str_alloc(t);
  free(t);
  return r;
}
struct str str_alloc (const char *r) {
  size_t l = strlen(r);
  char *d = alloc(l);
  memcpy(d, r, l);
  return (struct str) {d, l};
}
void str_free (struct str s) {
  free(s.d);
}

int main () {
  str_free((struct str) path_tempDirectory());
}
