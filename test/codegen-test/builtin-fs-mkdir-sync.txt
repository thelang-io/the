bool fs_isDirectorySync (struct str s) {
  char *c = alloc(s.l + 1);
  memcpy(c, s.d, s.l);
  c[s.l] = '\0';
  bool b = false;
  #ifdef THE_OS_WINDOWS
    struct _stat sb;
    if (_stat(c, &sb) == 0) {
      b = (sb.st_mode & _S_IFMT) == _S_IFDIR;
    }
  #else
    struct stat sb;
    if (stat(c, &sb) == 0) {
      b = (sb.st_mode & S_IFMT) == S_IFDIR;
    }
  #endif
  free(c);
  str_free(s);
  return b;
}
