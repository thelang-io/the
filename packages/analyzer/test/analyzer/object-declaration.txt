obj Test {}
obj Cat {
  mut name: str
  mut age: int
}

obj Dog {
  mut name: str
  mut age: int

  fn method1 () {}
  fn method2 (self: ref Self) {}
  fn method3 (mut self: ref Self) {}
  fn method4 (a: int) {}
  fn method5 (self: ref Self, b: int...) {}
  fn method6 (mut self: ref Self, c := 10) {}
}

main {
  v1 := Test{}

  v2 := Cat{}
  v3 := Cat{name: "Pav", age: 10}

  v4 := Dog{}
  mut v5 := Dog{name: "Lev", age: 12}
  v5.method1()
  v5.method2()
  v5.method3()
  v5.method4(10)
  v5.method5()
  v5.method5(0)
  v5.method5(0, 1)
  v5.method5(0, 1, 2)
  v5.method6()
  v5.method6(20)
}
